summary_soil_moisture <-  
    data_soil_moisture_cleaned %>% 
    group_by(treatment,nfixer,date_day_month,sm_measured ) %>% 
    
    summarise(mean_sm = mean(soil_moisture),
              sd_sm   = sd(soil_moisture)) %>% 
    
    arrange(nfixer) %>% 
    arrange(date_day_month)
