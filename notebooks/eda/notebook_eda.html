Table Plant’s performace measures

data_complete %>% 
  select(-id) %>% 
  select(treatment, nfixer,
         
         # Mass fractions
         rmf, smf, lmf, 
         
         # Dry weights 
         total_biomass, above_biomass,stem_dry_weight,
         root_dry_weight,whole_leaf_dry_weight) %>% 
  
  group_by(treatment, nfixer) %>% 
  diagnose_numeric() %>% 
  mutate_if(is.numeric, round, 2) %>% 
  datatable()

Boxplots Plant’s performace measures

plots_performance  <-
  vars %>%
      filter(V1 %in% c("rmf", "smf", "lmf",
                        # Dry weights 
                       "total_biomass", "above_biomass","stem_dry_weight",
                       "root_dry_weight", "whole_leaf_dry_weight")) %>% 
      
      # Gererate plots
      pmap( ~ boxplot_plot_pmap(data = data_complete,
                                y = !!sym(..1), x = !!sym(..2), 
                                fill = !!sym(..3)))

cowplot::plot_grid(plotlist = plots_performance, ncol = 2)
